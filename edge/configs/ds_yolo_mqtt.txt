################################################################################
# ds_yolo_mqtt.txt — 4 file sources -> YOLO11n -> tracker -> MQTT (headless).
#
# Adapted from deepstream/work/mqtt/test5_mqtt.txt. Changes: 4 sources,
# batch-size 4, YOLO11 nvinfer config, fakesink instead of a window,
# file-loop so the demo keeps publishing.
#
# Run: docker compose up   (see edge/README.md)
################################################################################

[application]
enable-perf-measurement=1
perf-measurement-interval-sec=5

# 2x2 composite of the 4 streams — what the file and RTSP sinks show.
[tiled-display]
enable=1
rows=2
columns=2
width=1280
height=720
gpu-id=0
nvbuf-memory-type=0

# One section per stream so any of them can be swapped for a different
# file/RTSP uri later without touching the rest.
[source0]
enable=1
type=3
uri=file:///videos/construction_1.mp4
num-sources=1
gpu-id=0
nvbuf-memory-type=0

[source1]
enable=1
type=3
uri=file:///videos/construction_2.mp4
num-sources=1
gpu-id=0
nvbuf-memory-type=0

[source2]
enable=1
type=3
uri=file:///videos/construction_3.mp4
num-sources=1
gpu-id=0
nvbuf-memory-type=0

[source3]
enable=1
type=3
uri=file:///videos/construction_4.mp4
num-sources=1
gpu-id=0
nvbuf-memory-type=0

# Output video (tiled view with boxes/labels). MKV stays playable even if
# the container is killed before a clean EOS.
[sink0]
enable=1
#Type - 1=FakeSink 2=EglSink 3=File 4=UDPSink/RTSP
type=3
#1=mp4 2=mkv
container=2
#1=h264 2=h265
codec=1
#encoder type 0=Hardware 1=Software
enc-type=0
sync=0
bitrate=4000000
output-file=/out/output.mkv
source-id=0
gpu-id=0
nvbuf-memory-type=0

# Live view for demos: rtsp://localhost:8554/ds-test (VLC or ffplay).
# sync=1 throttles the whole pipeline to real time so the stream is watchable.
[sink2]
enable=1
type=4
codec=1
enc-type=0
sync=1
bitrate=4000000
profile=0
rtsp-port=8554
udp-port=5400
gpu-id=0
nvbuf-memory-type=0

# Convert metadata to JSON and publish it over MQTT.
[sink1]
enable=1
#Type - 1=FakeSink 2=EglSink 3=File 4=UDPSink 5=nvdrmvideosink 6=MsgConvBroker
type=6
msg-conv-config=/configs/msgconv_config.txt
#(0)=full DeepStream schema  (1)=minimal schema (smaller JSON, used here)
msg-conv-payload-type=1
# MQTT proto adapter (ships in the image) instead of the default Kafka one.
msg-broker-proto-lib=/opt/nvidia/deepstream/deepstream/lib/libnvds_mqtt_proto.so
# "<host>;<port>" — reachable as service "mosquitto" on the compose network.
msg-broker-conn-str=mosquitto;1883
# Broker credentials/TLS (ignored under anonymous, but this is where they go).
msg-broker-config=/configs/cfg_mqtt.txt
topic=ds/detections

# Draw boxes, labels, and track IDs on the video sinks.
[osd]
enable=1
# CPU draw mode: the GPU mode draws skewed/tilted rectangles on this setup
# and cannot render text at all.
process-mode=0
gpu-id=0
border-width=1
text-size=15
text-color=1;1;1;1
text-bg-color=0.3;0.3;0.3;1
font=Arial
show-clock=0
nvbuf-memory-type=0

[streammux]
gpu-id=0
live-source=0
batch-size=4
batched-push-timeout=40000
width=1920
height=1080
nvbuf-memory-type=0

[primary-gie]
enable=1
gpu-id=0
batch-size=4
nvbuf-memory-type=0
interval=0
gie-unique-id=1
config-file=/configs/config_infer_primary_yolo11.txt

[tracker]
enable=1
tracker-width=960
tracker-height=544
ll-lib-file=/opt/nvidia/deepstream/deepstream/lib/libnvds_nvmultiobjecttracker.so
ll-config-file=/opt/nvidia/deepstream/deepstream/samples/configs/deepstream-app/config_tracker_NvDCF_perf.yml
gpu-id=0
display-tracking-id=0

[tests]
# Loop the files forever so events keep flowing.
file-loop=1
